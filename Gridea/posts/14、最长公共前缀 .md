class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        if not strs :
            return ''
        a = (min(strs, key=lambda i: len(i)))
        n = len(a)
        if n==0 :
            return a
        for i in range(n):
            c=strs[0][i]
            for ele in strs:
                if ele[i]==c:
                    continue
                else:
                        break
            else:
                continue
            return strs[0][:i]
        return  strs[0][:i+1]
